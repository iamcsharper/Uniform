# Backdrop

Компонент для затемнения экрана при открытии модальных окон

```js
import Backdrop from '@controls/future/Backdrop';

const [isOpen, setOpen] = useState(false);

return (
    <Portal>
        <Backdrop open={isOpen} onClose={() => setOpen(false)} />
    </Portal>
);
```

<Story
    name="Basic"
    args={{
        invisible: false,
    }}
>
    {(args) => {
        const [isOpen, setOpen] = useState(false);
        return (
            <div style={{ zIndex: 1, position: 'relative' }}>
                <Button onClick={() => setOpen(!isOpen)}>{isOpen ? 'Close' : 'Open'} me</Button>
                <Backdrop
                    {...args}
                    open={isOpen}
                    onClose={e => {
                        setOpen(false);
                        action(e)();
                    }}
                >
                    Dummy content in background
                </Backdrop>
            </div>
        );
    }}
</Story>
